实现堆排序的动态演示，直接创建heap对象；之后，将待排序的元素通过putVal()接口添加；然后调用sort()接口，进行排序；最后通过present()接口，实现动态演示的调用。 
	
	heap.create(15); //创建最多含有15个元素的heap
	var num = [12, 5, 34, 1, 6, 23, 89, 37, 56, 80, 64, 72, 50, 99, 100];

	for (var i = 0; i < num.length - 1; ++i) //添加元素
		heap.putVal(num[i]);

	heap.sort(); //调用排序接口

	heap.present(); ////调用动态演示接口
