实现希尔排序的动态演示，直接创建shell对象；之后，将待排序的元素通过putVal()接口添加；然后调用sort()接口，进行排序；最后通过present()接口，实现动态演示的调用。 
	
	shell.create(15); //创建最多含有8个元素的selection
	var num = [12, 5, 34, 1, 6, 23, 89, 37, 56, 80, 64, 72, 50, 99, 100];

	for (var i = 0; i < num.length; ++i) //添加元素
		shell.putVal(num[i]);

	shell.sort(); //调用排序接口
	shell.present(); //调用动态演示接口
