实现二分查找的动态演示，仅需创建bs对象，之后通过putVal()接口，添加元素，通过delVal()接口，删除元素，或通过getVal()接口，查找元素。 
	
	bs.create(11); //创建的最多可容纳11个元素的bs对象
	var num = [5, 13, 19, 92, 21, 37, 75, 64, 80, 56, 88];

	for (var i = 0; i < 9; ++i) //添加元素
		bs.putVal(num[i]);

	bs.getVal(92); //查找元素

	bs.putVal(37); //添加元素

	bs.getVal(88);

	bs.putVal(9); //添加元素

	bs.delVal(19); //删除元素
	bs.delVal(17);

	bs.present(); //调用动态演示接口
